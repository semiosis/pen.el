** elisp
#+BEGIN_SRC emacs-lisp -n :async :results verbatim code
  (defun pen-find-file (path)
    "Create directories and edit file"
    (pen-snc (cmd "mkdir" "-p" (f-dirname path)))
    (if (re-match-p "/$" path)
        (progn
          (pen-snc (cmd "mkdir" "-p" path))
          (find-file path))
      (find-file path)))
#+END_SRC
